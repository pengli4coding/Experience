数据库（database）的增删改查

增：
create database DB_A;
删：
drop database DB_A;
查：
show databases;



表（table）的增删改查

增：
create table TABLE_A
(
EMP_NO int not null,
FIRST_NAME varchar(14) not null,
LAST_NAME varchar(14) not null,
BIRTHDAY date not null,
primary key(EMP_NO)
)engine=innodb default charset=utf8
删：
drop table TABLE_A;
查：
show tables;--查询当前数据库中所有的表
desc TABLE_A;--查询表TABLE_A的表结构
show create table TABLE_A;--查询创建表TABLE_A的建表脚本语句
