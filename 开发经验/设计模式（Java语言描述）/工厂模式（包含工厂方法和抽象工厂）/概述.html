先问自己一个问题，为什么要用工厂模式？

观察以下代码，这是我们不用工厂模式的时候的写法：

        
// Dao接口，定义了增删改查方法
public interface Dao {
	void creat();
	void retrieve();
	void update();
	void delete();
}

// 实现了Dao接口，数据库类型为mysql的增删改查操作
public class DaoMysqlImpl implements Dao{
	@Override
	public void creat() {
		System.out.println("往mysql数据库中增加数据");
	}
	@Override
	public void retrieve() {
		System.out.println("往mysql数据库中获取数据");
	}
	@Override
	public void update() {
		System.out.println("往mysql数据库中更新数据");
	}
	@Override
	public void delete() {
		System.out.println("往mysql数据库中删除数据");
	}

}

// 实现了Dao接口，数据库类型为oracle的增删改查操作
public class DaoOracleImpl implements Dao{
	@Override
	public void creat() {
		System.out.println("往oracle数据库中增加数据");
	}
	@Override
	public void retrieve() {
		System.out.println("往oracle数据库中获取数据");
	}
	@Override
	public void update() {
		System.out.println("往oracle数据库中更新数据");
	}
	@Override
	public void delete() {
		System.out.println("往oracle数据库中删除数据");
	}

}

// 增删改查操作类
public class CRUD {
	
	private Dao dao = new DaoMysqlImpl();// 这里就是最大的问题，需要new对象出来
	
	public void dbCRUD() {
		dao.creat();
		dao.retrieve();
		dao.update();
		dao.delete();
	}
	
	public static void main(String[] args) {
		CRUD crud = new CRUD();
		crud.dbCRUD();
	}
}
