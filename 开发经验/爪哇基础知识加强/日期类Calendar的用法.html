// 由于Calendar是抽象类，无法通过new来获取实例，而是通过Calendar.getInstance()来获取Calendar类的实例

Calendar ca = Calendar.getInstance();

// 获取当前时间的年、月、日、时、分、秒

int year = ca.get(Calendar.YEAR);// 获取当前年份

int month = ca.get(Calendar.MONTH)+1;// 获取当前月份（注意，这里的月份范围是0~11，所以要获取当前月的话需要加一）

int day = ca.get(Calendar.DAY_OF_MONTH);// 获取当前日

int hour = ca.get(Calendar.HOUR_OF_DAY);// 获取当前时（注意，这里是24小时制的显示）

int minute = ca.get(Calendar.MINUTE);// 获取当前分

int second = ca.get(Calendar.SECOND);// 获取当前秒
